1计算并输出s的值。s的计算方法见下式，其中m为实数，其值由键盘读入。计算时，要求最后一项的绝对值小于1e-4，输出结果保留两位小数。

#include<stdio.h>
#include<math.h>
int main()
{
int n=1,sign=1;
double s=0,m,x=1;
scanf("%lf",&m);
do
{ x=x*m/n; s+=sign*x;
sign=-sign;
n++;
}while(fabs(x)>=1e-4);
printf("%.2f\n",s);
return 0;
}
2以每行8个的形式输出100-999内的质数

#include<stdio.h>
int main()
{ int m,n,i=0;
for(m=100;m<=999;m++)
{
for(n=2;n<m;n++)
{if(m%n==0)
break;
}
if(m==n)
{printf("%4d",m);
i++;
if(i%8==0)
printf("\n");
}
}
printf("\n");
return 0;
}
